    [cc lang="lisp"]
    (defn addup
    ([limit] (addup limit 0 0))
    ([limit current sum]
    (if (< limit current)
    sum
    (addup limit (+ 1 current) (+ current sum)))))
    [/cc]

    调用

    (addup 10)

    可以得到55

    (addup 50000)
